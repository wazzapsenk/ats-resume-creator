% Custom commands
\newcommand{\resumeheader}[6]{
  \begin{center}
    {\Large\bfseries #1 #2} \\
    \vspace{2pt}
    #3 \\
    \vspace{2pt}
    #4 \hspace{5pt} $\cdot$ \hspace{5pt} #5 \\
    {% if #6 %}
    \vspace{2pt}
    #6
    {% endif %}
  \end{center}
  \vspace{10pt}
}

\newcommand{\experienceitem}[5]{
  \subsection{#1 \hfill #2}
  \textit{#3} \hfill \textit{#4}
  {% if #5 %}
  \vspace{2pt}

  #5
  {% endif %}
  \vspace{5pt}
}

\newcommand{\educationitem}[5]{
  \subsection{#1 \hfill #2}
  \textit{#3} \hfill \textit{#4}
  {% if #5 %}

  #5
  {% endif %}
  \vspace{5pt}
}

\newcommand{\projectitem}[3]{
  \subsection{#1{% if #2 %} \hfill \href{#2}{#2}{% endif %}}
  #3
  \vspace{5pt}
}

\newcommand{\skillsection}[2]{
  \textbf{#1:} #2 \\
  \vspace{2pt}
}
